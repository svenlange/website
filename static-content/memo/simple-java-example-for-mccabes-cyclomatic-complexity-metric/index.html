            // Node 1
            return;
        }
        if (b) {
            // Node 2
            if (d) {
                // Node 3
                for (int i = 0; i < 10; i++) {
                    // Node 4
                }
            } else {
                // Node 5
            }
        } else if (c) {
            // Node 6
        } else {
            // Node 7
        }
    }
}

pmd and checkstyle say that a complexity of 1 to 4 is ok, 5 to ? acceptable and everything above should be improved.
